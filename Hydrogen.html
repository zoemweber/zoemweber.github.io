2D Incompressible Fluid Flow

# IMPORT OUR NORMAL STUFF
from numpy import *
from matplotlib.pyplot import *
%matplotlib inline

f = fluid.FluidFlow2D(ncells=80, xmin=-50, xmax=50,
  solver_tolerence=3.e-4)

# gets the coordinate arrays
xx, yy = f.get_coords()

# setup an initial condition
vxo = zeros_like(xx)
vyo = zeros_like(xx)
dno = ones_like(xx)

# set higher density on the top

dno[yy>5*sin(xx/6)-30] = 2
dno[yy>5*sin(xx/6)+30] = 1

#vxo[]
vxo[yy>5*sin(xx/6)-30] = 2
vxo[yy>5*sin(xx/6)+30] = 0

# send this initial condition to the code
diff_coeff = 1.e-3
f.setup_initial_condition(vxo, vyo, dno, Diff_Coeff=diff_coeff,bctype="wrap")

vxlist = []
vylist = []
dnlist = []

f.evolve_fluid(vxlist,vylist,dnlist,tmax=400,plot_every=10,save_every=10,bctype="wrap")
f.make_monitoring_plots(velocity_vectors = True)
